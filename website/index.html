How FormBridge Works

1

HTML Form

User submits form with name, email, message

2

API Gateway

Request validated, API key checked, rate limited

3

Lambda Handler

Process form data, validate, prepare email

4

Store & Email

Save to DynamoDB, send email via SES

curl -X POST https://api.formbridge.dev/submit \
  -H "Content-Type: application/json" \
  -H "X-Api-Key: your_api_key" \
  -d '{
    "name": "John Doe",
    "email": "john@example.com",
    "message": "Hello FormBridge!"
  }'

# Response: { "id": "sub_1234...", "status": "sent" }
const response = await fetch('https://api.formbridge.dev/submit', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-Api-Key': 'your_api_key'
  },
  body: JSON.stringify({
    name: 'John Doe',
    email: 'john@example.com',
    message: 'Hello FormBridge!'
  })
});

const data = await response.json();
console.log('Submission ID:', data.id);
import requests

response = requests.post(
  'https://api.formbridge.dev/submit',
  json={
    'name': 'John Doe',
    'email': 'john@example.com',
    'message': 'Hello FormBridge!'
  },
  headers={
    'X-Api-Key': 'your_api_key'
  }
)

print('Submission ID:', response.json()['id'])
